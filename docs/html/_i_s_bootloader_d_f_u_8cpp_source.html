69int cISBootloaderDFU::get_num_devices()
70{
71 int present = 0;
72
73 libusb_device** device_list;
74 libusb_device* dev;
75 //libusb_device_handle* dev_handle;
76 struct libusb_device_descriptor desc;
77 struct libusb_config_descriptor* cfg;
78 int ret_libusb;
79
80 if(m_DFUmutex.try_lock())
81 {
82 libusb_init(NULL);
83
84 size_t device_count = libusb_get_device_list(NULL, &device_list);
85
86 for (size_t i = 0; i < (device_count - 1); ++i) {
87 dev = device_list[i];
88
89 ret_libusb = libusb_get_device_descriptor(dev, &desc);
90 if(ret_libusb < 0) continue;
91
92 // Check vendor and product ID
93 if (desc.idVendor != STM32_DESCRIPTOR_VENDOR_ID) continue; // must be some other usb device
94 if (desc.idProduct != STM32_DESCRIPTOR_PRODUCT_ID) continue; // must be some other usb device
95
96 ret_libusb = libusb_get_config_descriptor(device_list[i], 0, &cfg);
97
98 // USB-IF DFU interface class numbers
99 if(cfg->interface->altsetting[0].bInterfaceClass != 0xFE ||
100 cfg->interface->altsetting[0].bInterfaceSubClass != 0x01 ||
101 cfg->interface->altsetting[0].bInterfaceProtocol != 0x02
102 ) continue;
103
104 present++;
105 }
106
107 libusb_free_device_list(device_list, 1);
108
109 libusb_exit(NULL);
110
111 m_DFUmutex.unlock();
112 }
113
114 return present;
115}

179is_operation_result cISBootloaderDFU::get_serial_number_libusb(libusb_device_handle** handle, uint32_t& sn, std::string& uidstr, uint8_t sn_idx)
180{
181 dfu_status status;
182 uint32_t waitTime = 0;
183 dfu_state state;
184 uint8_t stringIdx;
185 dfu_error ret_dfu;
186
187 int ret_libusb;
188
189 uint8_t rxBuf[1024] = {0};
190
191 SLEEP_MS(100);
192
193 // Reset the device
194 ret_libusb = libusb_reset_device(*handle);
195 if (ret_libusb < LIBUSB_SUCCESS) { return IS_OP_ERROR; }
196
197 ret_libusb = libusb_claim_interface(*handle, 0);
198 if (ret_libusb < LIBUSB_SUCCESS) { return IS_OP_ERROR; }
199
200 // Cancel any existing operations
201 ret_libusb = dfu_ABORT(handle);
202 if (ret_libusb < LIBUSB_SUCCESS) { return IS_OP_ERROR; }
203
204 // Reset status to good
205 ret_dfu = dfu_wait_for_state(handle, DFU_STATE_IDLE);
206 if (ret_dfu < DFU_ERROR_NONE) { return IS_OP_ERROR; }
207
208 // Get the string containing the serial number from the device
209 unsigned char uid[IS_DFU_UID_MAX_SIZE];
210 ret_libusb = libusb_get_string_descriptor_ascii(*handle, sn_idx, uid, sizeof(uid));
211 if(ret_libusb < LIBUSB_SUCCESS) uid[0] = '\0'; // Set the serial number as none
212
213 uidstr = std::string((const char*)uid);
214
215 // Get the 1K OTP section from the chip
216 // 0x1FFF7000 is the address. Little endian.
217 {
218 // Set the address pointer (command is 0x21)
219 uint8_t txBuf[] = { 0x21, 0x00, 0x70, 0xFF, 0x1F };
220 ret_libusb = dfu_DNLOAD(handle, 0, txBuf, sizeof(txBuf));
221 if(ret_libusb < LIBUSB_SUCCESS) return IS_OP_ERROR;
222
223 // Address pointer takes effect after GETSTATUS command
224 ret_libusb = dfu_GETSTATUS(handle, &status, &waitTime, &state, &stringIdx);
225 if(ret_libusb < LIBUSB_SUCCESS || status != DFU_STATUS_OK || state != DFU_STATE_DNBUSY) return IS_OP_ERROR;
226 ret_libusb = dfu_GETSTATUS(handle, &status, &waitTime, &state, &stringIdx);
227 if(ret_libusb < LIBUSB_SUCCESS || status != DFU_STATUS_OK) return IS_OP_ERROR;
228
229 // Get out of download mode
230 dfu_ABORT(handle);
231
232 // Read the full OTP page
233 ret_libusb = dfu_UPLOAD(handle, 2, rxBuf, 1024);
234 if(ret_libusb < LIBUSB_SUCCESS) return IS_OP_ERROR;
235 }
236
237 int index = 0;
238 uint8_t* otp_mem = (uint8_t*)rxBuf;
239
240 // Look for the first section of zeroes
241 uint8_t cmp[OTP_SECTION_SIZE];
242 memset(cmp, 0xFF, OTP_SECTION_SIZE);
243 bool foundSn = true;
244 while(memcmp(cmp, otp_mem, OTP_SECTION_SIZE) != 0)
245 {
246 otp_mem += OTP_SECTION_SIZE; index++;
247 if(index >= OTP_NUM_SECTIONS)
248 {
249 foundSn = false; break; // No more room in OTP
250 }
251 }
252
253 // Go back one, to the last filled section
254 index--;
255 is_dfu_otp_id_t* id = (is_dfu_otp_id_t*)((index * OTP_SECTION_SIZE) + rxBuf);
256
257 uint64_t key = OTP_KEY;
258 if(memcmp(otp_mem - 8, &key, 8) == 0 && foundSn)
259 {
260 sn = id->serialNumber;
261
262 libusb_release_interface(*handle, 0);
263 return IS_OP_OK;
264 }
265
266 return IS_OP_ERROR;
267}

277is_operation_result cISBootloaderDFU::download_image(std::string filename)
278{
279 int ret_libusb;
280 dfu_error ret_dfu;
281 ihex_image_section_t image[MAX_NUM_IHEX_SECTIONS];
282 size_t image_sections;
283
284 // Reset the device
285 ret_libusb = libusb_reset_device(m_dfu.handle_libusb);
286 if(ret_libusb < LIBUSB_SUCCESS)
287 {
288 return IS_OP_ERROR;
289 }
290
291 SLEEP_MS(100);
292
293 ret_libusb = libusb_claim_interface(m_dfu.handle_libusb, 0);
294 if (ret_libusb < LIBUSB_SUCCESS) { return IS_OP_ERROR; }
295
296 // Cancel any existing operations
297 ret_libusb = dfu_ABORT(&m_dfu.handle_libusb);
298 if (ret_libusb < LIBUSB_SUCCESS) { return IS_OP_ERROR; }
299
300 // Reset status to good
301 ret_dfu = dfu_wait_for_state(&m_dfu.handle_libusb, DFU_STATE_IDLE);
302 if (ret_dfu < DFU_ERROR_NONE) { return IS_OP_ERROR; }
303
304 // Load the firmware image
305 image_sections = ihex_load_sections(filename.c_str(), image, MAX_NUM_IHEX_SECTIONS);
306 if(image_sections <= 0) { return IS_OP_ERROR; }
307
308 int image_total_len = 0;
309 for(size_t i = 0; i < image_sections; i++)
310 {
311 image_total_len += image[i].len;
312 }
313
314 // If starting address is 0, set it to 0x08000000 (start of flash memory)
315 if(image[0].address == 0x00000000)
316 {
317 for(size_t i = 0; i < image_sections; i++)
318 {
319 image[i].address += 0x08000000;
320 }
321 }
322
323 uint32_t bytes_written_total = 0;
324
325 status_update("(DFU) Erasing flash...", IS_LOG_LEVEL_INFO);
326
327 // Erase memory (only erase pages where firmware lives)
328 for(size_t i = 0; i < image_sections; i++)
329 {
330 if(image[i].address & STM32_PAGE_ERROR_MASK)
331 {
332 continue; // Page is not aligned with write location
333 }
334
335 if(image[i].image == NULL || image[i].len == 0)
336 {
337 continue; // Null image
338 }
339
340 uint32_t byteInSection = 0;
341
342 do {
343 uint32_t pageAddress = byteInSection + image[i].address;
344 uint8_t eraseCommand[5];
345
346 eraseCommand[0] = 0x41;
347 memcpy(&eraseCommand[1], &pageAddress, 4);
348
349 ret_libusb = dfu_DNLOAD(&m_dfu.handle_libusb, 0, eraseCommand, 5);
350 // if (ret_libusb < LIBUSB_SUCCESS)
351 // {
352 // ihex_unload_sections(image, image_sections);
353 // return IS_OP_ERROR;
354 // }
355
356 ret_dfu = dfu_wait_for_state(&m_dfu.handle_libusb, DFU_STATE_DNLOAD_IDLE);
357 // if (ret_dfu < DFU_ERROR_NONE)
358 // {
359 // ihex_unload_sections(image, image_sections);
360 // return IS_OP_ERROR;
361 // }
362
363 byteInSection += STM32_PAGE_SIZE;
364 bytes_written_total += STM32_PAGE_SIZE;
365
366 m_update_progress = 0.25f * ((float)bytes_written_total / (float)image_total_len);
367 m_update_callback(this, m_update_progress);
368 } while(byteInSection < image[i].len - 1);
369 }
370
371 bytes_written_total = 0;
372
373 status_update("(DFU) Programming flash...", IS_LOG_LEVEL_INFO);
374
375 // Write memory
376 for(size_t i = 0; i < image_sections; i++)
377 {
378 ret_dfu = dfu_wait_for_state(&m_dfu.handle_libusb, DFU_STATE_DNLOAD_IDLE);
379
380 ret_dfu = dfu_set_address_pointer(&m_dfu.handle_libusb, image[i].address);
381 if (ret_dfu < DFU_ERROR_NONE)
382 {
383 ihex_unload_sections(image, image_sections);
384 return IS_OP_ERROR;
385 }
386
387 uint32_t byteInSection = 0;
388
389 do {
390 uint8_t payload[STM32_PAGE_SIZE] = { 0 };
391 uint32_t payloadLen = STM32_PAGE_SIZE;
392 uint32_t bytesRemaining = image[i].len - byteInSection;
393 if (payloadLen > bytesRemaining)
394 {
395 payloadLen = bytesRemaining;
396 }
397
398 // Copy image into buffer for transmission
399 memset(payload, 0xFF, STM32_PAGE_SIZE);
400 memcpy(payload, &image[i].image[byteInSection], payloadLen);
401
402 uint8_t blockNum = (uint8_t)(byteInSection / STM32_PAGE_SIZE);
403
404 ret_libusb = dfu_DNLOAD(&m_dfu.handle_libusb, blockNum + 2, payload, STM32_PAGE_SIZE);
405 if (ret_libusb < LIBUSB_SUCCESS)
406 {
407 ihex_unload_sections(image, image_sections);
408 return IS_OP_ERROR;
409 }
410
411 ret_dfu = dfu_wait_for_state(&m_dfu.handle_libusb, DFU_STATE_DNLOAD_IDLE);
412 if (ret_dfu < DFU_ERROR_NONE)
413 {
414 ihex_unload_sections(image, image_sections);
415 return IS_OP_ERROR;
416 }
417
418 byteInSection += payloadLen;
419 bytes_written_total += payloadLen;
420
421 m_update_progress = 0.25f + 0.75f * ((float)bytes_written_total / (float)image_total_len);
422 m_update_callback(this, m_update_progress);
423 } while (byteInSection < image[i].len - 1);
424 }
425
426 // Unload the firmware image
427 ihex_unload_sections(image, image_sections);
428
429 // Cancel any existing operations
430 ret_libusb = dfu_ABORT(&m_dfu.handle_libusb);
431 if (ret_libusb < LIBUSB_SUCCESS) { return IS_OP_ERROR; }
432
433 // Reset status to good
434 ret_dfu = dfu_wait_for_state(&m_dfu.handle_libusb, DFU_STATE_IDLE);
435 if (ret_dfu < DFU_ERROR_NONE) { return IS_OP_ERROR; }
436
437 libusb_release_interface(m_dfu.handle_libusb, 0);
438
439 return IS_OP_OK;
440}

484is_operation_result cISBootloaderDFU::reboot()
485{
486 int ret_libusb;
487 dfu_error ret_dfu;
488
489 // Cancel any existing operations
490 ret_libusb = dfu_ABORT(&m_dfu.handle_libusb);
491 if (ret_libusb < LIBUSB_SUCCESS) return IS_OP_ERROR;
492
493 // Reset status to good
494 ret_dfu = dfu_wait_for_state(&m_dfu.handle_libusb, DFU_STATE_IDLE);
495 if (ret_dfu < DFU_ERROR_NONE) return IS_OP_ERROR;
496
497 // Set address pointer to flash
498 ret_dfu = dfu_set_address_pointer(&m_dfu.handle_libusb, 0x08000000);
499 if (ret_dfu < DFU_ERROR_NONE) return IS_OP_ERROR;
500
501 // Request DFU leave
502 ret_libusb = dfu_DNLOAD(&m_dfu.handle_libusb, 0, NULL, 0);
503 if (ret_libusb < LIBUSB_SUCCESS) return IS_OP_ERROR;
504
505 // Execute DFU leave
506 ret_dfu = dfu_wait_for_state(&m_dfu.handle_libusb, DFU_STATE_MANIFEST);
507 if (ret_dfu < DFU_ERROR_NONE) return IS_OP_ERROR;
508
509 // Reset USB device
510 ret_libusb = libusb_reset_device(m_dfu.handle_libusb);
511 if (ret_libusb < LIBUSB_SUCCESS) return IS_OP_ERROR;
512
513 return IS_OP_OK;
514}

516int cISBootloaderDFU::dfu_GETSTATUS(libusb_device_handle** dev_handle, dfu_status* status, uint32_t* delay, dfu_state* state, uint8_t* i_string)
517{
518 int ret_libusb;
519 uint8_t buf[6] = { 0 };
520
521 ret_libusb = libusb_control_transfer(*dev_handle, 0b10100001, 0x03, 0, 0, buf, 6, 100);
522
523 *status = (dfu_status)buf[0];
524 *delay = (buf[3] << 16) | (buf[2] << 8) | buf[1];
525 *state = (dfu_state)buf[4];
526 *i_string = buf[5];
527
528 return ret_libusb;
529}

574cISBootloaderDFU::dfu_error cISBootloaderDFU::dfu_wait_for_state(libusb_device_handle** dev_handle, dfu_state required_state)
575{
576 int ret_libusb;
577
578 dfu_status status;
579 uint32_t waitTime = 0;
580 dfu_state state;
581 uint8_t stringIndex;
582
583 uint8_t tryCounter = 0;
584
585 ret_libusb = dfu_GETSTATUS(dev_handle, &status, &waitTime, &state, &stringIndex);
586 if (ret_libusb < LIBUSB_SUCCESS)
587 {
588 return DFU_ERROR_LIBUSB;
589 }
590
591 while (status != DFU_STATUS_OK || state != required_state)
592 {
593 if (status != DFU_STATUS_OK)
594 {
595 ret_libusb = dfu_CLRSTATUS(dev_handle);
596 if (ret_libusb < LIBUSB_SUCCESS) return DFU_ERROR_LIBUSB;
597 }
598
599 SLEEP_MS(_MAX(waitTime, 10)); // TODO: Windows
600
601 ret_libusb = dfu_GETSTATUS(dev_handle, &status, &waitTime, &state, &stringIndex);
602 if (ret_libusb < LIBUSB_SUCCESS)
603 {
604 return DFU_ERROR_LIBUSB;
605 }
606
607 tryCounter++;
608 if (tryCounter > 4) return DFU_ERROR_TIMEOUT;
609 }
610
611 return DFU_ERROR_NONE;
612}
