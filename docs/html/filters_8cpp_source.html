36void init_iir_filter(iif_filter_t *f)
37{
38 int gama;
39 int alpha;
40 float TsFc;
41
42 if(f->opt.n_channels > MAX_NUMBER_IIR_CHANNELS)
43 {
44// dg_printf("IIR channels exceeded max number: %d. Consider increasing max number", MAX_NUMBER_IIR_CHANNELS);
45// exit(1);
46 }
47
48 f->opt.bit_shift = (ACCUM_WORD_NBITS-1-f->opt.sig_word_nbits)/2;
49
50 // Gama = adc_to_iir = alpha + beta
51 gama = 1<<f->opt.bit_shift;
52 TsFc = f->opt.Fc/f->opt.Fs;
53 alpha = (int)( TsFc/(1.0 + TsFc) * (float)gama );
54 f->opt.beta = gama - alpha;
55 // alpha_x allows us to combine the adc_to_iir and alpha multiplies into one step
56 f->opt.alpha_x = alpha<<f->opt.bit_shift;
57 f->opt.iir_to_adc = 1.0f/(float)gama;
58}

61void iir_filter_u16(iif_filter_t *f, unsigned short input[], float output[])
62{
63 unsigned int i,j;
64
65 for (j=0; j<f->opt.input_size; j+=f->opt.n_channels)
66 {
67 for (i=0; i<f->opt.n_channels; i++)
68 {
69 f->accum[i] = f->opt.beta*f->accum[i] + f->opt.alpha_x*((int)input[j+i]);
70 f->accum[i] >>= f->opt.bit_shift;
71 }
72 }
73
74 for (i=0; i<f->opt.n_channels; i++)
75 output[i] = f->opt.iir_to_adc*(float)f->accum[i];
76}

106void running_mean_filter( float input[], float mean[], int arraySize, int sampleCount )
107{
108 int i;
109 float alpha;
110
111 if( sampleCount == 0 )
112 alpha = 1.0f;
113 else
114 alpha = 1.0f / (float)sampleCount;
115
116 // Find running average
117 for( i=0; i<arraySize; i++ )
118 mean[i] = (1.0f-alpha)*mean[i] + (alpha)*input[i];

161void recursive_moving_mean_var_filter(float *mean, float *var, float input, int sampleCount)
162{
163 float dx, div;
164
165 if (sampleCount <= 0) return;
166
167 dx = input - *mean;
168
169 // Expected moving average
170 div = 1.0f / (float)sampleCount;
171 *mean += dx * div;
172
173 // Expected moving variance
174 *var = ((float)(sampleCount * sampleCount - sampleCount - 1) * (*var) + (float)(sampleCount - 1) * dx * dx) * div * div;
175}

198int tripleToSingleImu(imu_t *result, const imu3_t *di)
199{
200 imu_t imu = {};
201 imu.time = di->time;
202 imu.status = di->status;
203
204 int cnt = 0;
205
206 for (int d=0; d<3; d++)
207 {
208 uint32_t imuOkBitMask = IMU_STATUS_IMU1_OK<<d;
209 if ((di->status&imuOkBitMask)==imuOkBitMask)
210 {
211 add_Vec3_Vec3(imu.I.pqr, imu.I.pqr, di->I[d].pqr);
212 add_Vec3_Vec3(imu.I.acc, imu.I.acc, di->I[d].acc);
213 cnt++;
214 }
215 }
216
217 if (cnt)
218 {
219 float div = 1.0f/(float)cnt;
220 mul_Vec3_X(imu.I.pqr, imu.I.pqr, div);
221 mul_Vec3_X(imu.I.acc, imu.I.acc, div);
222 }
223
224 *result = imu;
225 return cnt;
226}

258void tripleToSingleImuAxis(imu_t* result, const imu3_t* di, bool exclude_gyro[3], bool exclude_acc[3], int iaxis)
259{
260 float w = 0.0f, a = 0.0f;
261 int cnt_gyro = 0, cnt_acc = 0;
262
263 for (int idev = 0; idev < 3; idev++)
264 {
265 if (!exclude_gyro[idev])
266 {
267 w += di->I[idev].pqr[iaxis];
268 cnt_gyro++;
269 }
270 if (!exclude_acc[idev])
271 {
272 a += di->I[idev].acc[iaxis];
273 cnt_acc++;
274 }
275 }
276 if (cnt_gyro > 0) w = w / (float)cnt_gyro;
277 if (cnt_acc > 0) a = a / (float)cnt_acc;
278
279 result->I.pqr[iaxis] = w;
280 result->I.acc[iaxis] = a;
281 result->time = di->time;
282 result->status = di->status;
283}

353{
354 ixVector3 tmp3;
355 float dt = (float)(imu->time - imuLast->time);
356
357 // Use Riemann Sum integral
358
359 // IMU - Delta Theta
360 mul_Vec3_X( tmp3, imu->I.pqr, dt );
361 add_Vec3_Vec3( output->theta, output->theta, tmp3 );
362
363 // IMU - Delta Velocity
364 mul_Vec3_X( tmp3, imu->I.acc, dt );
365 add_Vec3_Vec3( output->vel, output->vel, tmp3 );
366
367 // Update history
368 *imuLast = *imu;
369
370 return dt;

398void integrateDeltaThetaVelBortz(ixVector3 theta, ixVector3 vel, imus_t *imu, imus_t *imuLast, float Nsteps, float dt)
399{
400 ixVector3 wb, ab, deltaW, deltaA, tmp1, tmp2, tmp3, tmp4;
401 float dti, Kw, mag_theta2, mag_theta4, Kw0;
402
403 // for jj = 1: Nint
404 // wb = W0 + (jj - 1) / Nint * (W1 - W0);
405 // ab = A0 + (jj - 1) / Nint * (A1 - A0);
406 // % Bortz's formula
407 // phi_dot = wb + 0.5 * cross(phi, wb) + 1 / 12 * cross(phi, cross(phi, wb));
408 // % Savage's formula (faster): approximates phi with gyro integral
409 // % phi_dot = Wi + 0.5 * cross(WintB, wb)
410 // phi = phi + phi_dot * dt / Nint;
411 // dv = dv + (ab + cross(phi, ab)) * dt / Nint;
412 // end
413
414 sub_Vec3_Vec3(deltaW, imu->pqr, imuLast->pqr);
415 sub_Vec3_Vec3(deltaA, imu->acc, imuLast->acc);
416 div_Vec3_X(deltaW, deltaW, Nsteps);
417 div_Vec3_X(deltaA, deltaA, Nsteps);
418 cpy_Vec3_Vec3(wb, imuLast->pqr);
419 cpy_Vec3_Vec3(ab, imuLast->acc);
420 dti = dt / Nsteps;
421
422 Kw0 = 1.0f / 12.0f;
423 for (int jj = 0; jj < Nsteps; jj++)
424 {
425 // Coning and sculling integrals
426 cross_Vec3(tmp1, theta, wb);
427 cross_Vec3(tmp2, theta, tmp1);
428 cross_Vec3(tmp3, theta, ab);
429 cross_Vec3(tmp4, theta, tmp3);
430 mag_theta2 = dot_Vec3(theta);
431 mag_theta4 = mag_theta2 * mag_theta2;
432 Kw = Kw0 + mag_theta2 / 720.0f + mag_theta4 / 30240.0f + mag_theta4 * mag_theta2 / 1209600.0f;
433 for (int i = 0; i < 3; i++) {
434 theta[i] += (wb[i] + 0.5f * tmp1[i] + Kw * tmp2[i]) * dti;
435 vel[i] += (ab[i] + tmp3[i] + 0.5f * tmp4[i]) * dti;
436 }
437 // Advance wb, ab (minor step)
438 add_Vec3_Vec3(wb, wb, deltaW);
439 add_Vec3_Vec3(ab, ab, deltaA);
440 }
441}

444/* Direct numerical integration of coning and sculling integrals using Bortz's formula */
445float deltaThetaDeltaVelBortz(pimu_t *output, imu_t *imu, imu_t *imuLast, int Nsteps)
446{
447 float dt = (float)(imu->time - imuLast->time);
448
449 // IMU
450 integrateDeltaThetaVelBortz(output->theta, output->vel, &(imu->I), &(imuLast->I), (float)Nsteps, dt);
451
452 // Update history
453 *imuLast = *imu;
454
455 return dt;
456}
