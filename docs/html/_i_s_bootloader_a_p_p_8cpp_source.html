43eImageSignature cISBootloaderAPP::check_is_compatible()
44{
45 serialPortFlush(m_port);
46
47 // Get DID_DEV_INFO from the uINS.
48 is_comm_instance_t comm;
49 uint8_t buffer[2048];
50 is_comm_init(&comm, buffer, sizeof(buffer));
51 int messageSize;
52
53 messageSize = is_comm_get_data(&comm, DID_DEV_INFO, 0, 0, 0);
54 for(int i = 0; i < 2; i++) // HACK: Send this twice. After leaving DFU mode, the serial port doesn't respond to the first request.
55 if (messageSize != serialPortWrite(m_port, comm.buf.start, messageSize))
56 {
57 //serialPortClose(m_port);
58 return IS_IMAGE_SIGN_NONE;
59 }
60 messageSize = is_comm_get_data(&comm, DID_EVB_DEV_INFO, 0, 0, 0);
61 if (messageSize != serialPortWrite(m_port, comm.buf.start, messageSize))
62 {
63 //serialPortClose(m_port);
64 return IS_IMAGE_SIGN_NONE;
65 }
66
67 protocol_type_t ptype;
68 int n = is_comm_free(&comm);
69 dev_info_t* dev_info = NULL;
70 dev_info_t* evb_dev_info = NULL;
71 uint32_t valid_signatures = 0;
72 if ((n = serialPortReadTimeout(m_port, comm.buf.start, n, 200)))
73 {
74 comm.buf.tail += n;
75 while ((ptype = is_comm_parse(&comm)) != _PTYPE_NONE)
76 {
77 if(ptype == _PTYPE_INERTIAL_SENSE_DATA)
78 {
79 switch(comm.dataHdr.id)
80 {
81 case DID_DEV_INFO:
82 dev_info = (dev_info_t*)comm.dataPtr;
83 m_sn = dev_info->serialNumber;
84 if(dev_info->hardwareVer[0] == 5)
85 {
86 valid_signatures |= IS_IMAGE_SIGN_UINS_5;
87 valid_signatures |= IS_IMAGE_SIGN_ISB_STM32L4;
88 }
89 else if (dev_info->hardwareVer[0] == 3 || dev_info->hardwareVer[0] == 4)
90 {
91 valid_signatures |= IS_IMAGE_SIGN_UINS_3_16K | IS_IMAGE_SIGN_UINS_3_24K;
92 valid_signatures |= IS_IMAGE_SIGN_ISB_SAMx70_16K | IS_IMAGE_SIGN_ISB_SAMx70_24K;
93 }
94 break;
95 case DID_EVB_DEV_INFO:
96 evb_dev_info = (dev_info_t*)comm.dataPtr;
97 if (evb_dev_info->hardwareVer[0] == 2)
98 {
99 valid_signatures |= IS_IMAGE_SIGN_EVB_2_16K | IS_IMAGE_SIGN_EVB_2_24K;
100 valid_signatures |= IS_IMAGE_SIGN_ISB_SAMx70_16K | IS_IMAGE_SIGN_ISB_SAMx70_24K;
101 }
102 break;
103 }
104 }
105 }
106 }
107
108 return (eImageSignature)valid_signatures;
109}

121is_operation_result cISBootloaderAPP::reboot_down(uint8_t major, char minor, bool force)
122{
123 (void)force;
124 (void)minor;
125 (void)major;
126
127 m_info_callback(this, "(APP) Rebooting down into ISB mode...", IS_LOG_LEVEL_INFO);
128
129 // In case we are in program mode, try and send the commands to go into bootloader mode
130 uint8_t c = 0;
131
132 for (size_t loop = 0; loop < 10; loop++)
133 {
134 if (!serialPortWriteAscii(m_port, "STPB", 4)) break; // If the write fails, assume the device is now in bootloader mode.
135 if (!serialPortWriteAscii(m_port, m_app.enable_command, 4)) break;
136 c = 0;
137 if (serialPortReadCharTimeout(m_port, &c, 13) == 1)
138 {
139 if (c == '$')
140 {
141 // done, we got into bootloader mode
142 break;
143 }
144 }
145 else serialPortFlush(m_port);
146 }
147
148 return IS_OP_OK;
149}

151uint32_t cISBootloaderAPP::get_device_info()
152{
153 serialPortFlush(m_port);
154
155 // Get DID_DEV_INFO from the uINS.
156 is_comm_instance_t comm;
157 uint8_t buffer[2048];
158 is_comm_init(&comm, buffer, sizeof(buffer));
159 int messageSize;
160
161 messageSize = is_comm_get_data(&comm, DID_DEV_INFO, 0, 0, 0);
162 for(int i = 0; i < 2; i++) // HACK: Send this twice. After leaving DFU mode, the serial port doesn't respond to the first request.
163 if (messageSize != serialPortWrite(m_port, comm.buf.start, messageSize))
164 {
165 // serialPortClose(&ctx->handle.port);
166 return 0;
167 }
168 messageSize = is_comm_get_data(&comm, DID_EVB_DEV_INFO, 0, 0, 0);
169 if (messageSize != serialPortWrite(m_port, comm.buf.start, messageSize))
170 {
171 // serialPortClose(&ctx->handle.port);
172 return 0;
173 }
174 messageSize = is_comm_get_data(&comm, DID_EVB_STATUS, 0, 0, 0);
175 if (messageSize != serialPortWrite(m_port, comm.buf.start, messageSize))
176 {
177 // serialPortClose(&ctx->handle.port);
178 return 0;
179 }
180
181 // Wait 10ms for messages to come back
182 serialPortSleep(m_port, 10);
183
184 protocol_type_t ptype;
185 int n = is_comm_free(&comm);
186 dev_info_t* dev_info = NULL;
187 dev_info_t* evb_dev_info = NULL;
188 evb_status_t* evb_status = NULL;
189 uint8_t evb_version[4];
190 if ((n = serialPortReadTimeout(m_port, comm.buf.start, n, 200)))
191 {
192 comm.buf.tail += n;
193 while ((ptype = is_comm_parse(&comm)) != _PTYPE_NONE)
194 {
195 if(ptype == _PTYPE_INERTIAL_SENSE_DATA)
196 {
197 switch(comm.dataHdr.id)
198 {
199 case DID_DEV_INFO:
200 dev_info = (dev_info_t*)comm.dataPtr;
201 memcpy(m_app.uins_version, dev_info->hardwareVer, 4);
202 m_sn = dev_info->serialNumber;
203 break;
204 case DID_EVB_DEV_INFO:
205 evb_dev_info = (dev_info_t*)comm.dataPtr;
206 memcpy(evb_version, evb_dev_info->hardwareVer, 4);
207 break;
208 case DID_EVB_STATUS:
209 evb_status = (evb_status_t*)comm.dataPtr;
210 if(evb_status->firmwareVer[0]) memcpy(m_app.evb_version, evb_version, 4);
211 else memset(m_app.evb_version, 0, 4);
212 break;
213 }
214 }
215 }
216 }
217
218 return m_sn;
219}
