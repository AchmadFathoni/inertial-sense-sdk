48eImageSignature cISBootloaderBase::get_hex_image_signature(std::string filename, uint8_t* major, char* minor)
49{
50 ihex_image_section_t image;
51 size_t sections = ihex_load_sections(filename.c_str(), &image, 1);
52 size_t image_type;
53
54 if(sections == 1) // Signature must be in the first section of the image
55 {
56 uint8_t *target_signature;
57
58 for(image_type = 0; image_type < IS_IMAGE_SIGN_NUM_BITS_USED; image_type++)
59 {
60 switch(1 << image_type)
61 {
62 case IS_IMAGE_SIGN_UINS_3_16K: target_signature = bootloaderRequiredSignature_uINS_3_16K; break;
63 case IS_IMAGE_SIGN_UINS_3_24K: target_signature = bootloaderRequiredSignature_uINS_3_24K; break;
64 case IS_IMAGE_SIGN_EVB_2_16K: target_signature = bootloaderRequiredSignature_EVB_2_16K; break;
65 case IS_IMAGE_SIGN_EVB_2_24K: target_signature = bootloaderRequiredSignature_EVB_2_24K; break;
66 case IS_IMAGE_SIGN_UINS_5: target_signature = bootloaderRequiredSignature_uINS_5; break;
67 case IS_IMAGE_SIGN_ISB_STM32L4: target_signature = bootloaderRequiredSignature_STM32L4_bootloader; break;
68 case IS_IMAGE_SIGN_ISB_SAMx70_24K: target_signature = bootloaderRequiredSignature_SAMx70_bootloader_24K; break;
69 default: continue;
70 }
71
72 size_t k = 0;
73 for(size_t j = 0; j < image.len; j++)
74 {
75 if(image.image[j] == target_signature[k]) k++; // Found the right char, continue
76 else k = 0; // Didn't find the right char, reset to beginning of search
77
78 if(k >= BOOTLOADER_SIGNATURE_SIZE)
79 {
80 // In the bootloader images, the version bytes are stored directly after the signature
81 // A third byte is added as a checksum of the two preceding bytes.
82 if(image.image[j + 1] + image.image[j + 2] == image.image[j + 3])
83 {
84 if(major) *major = image.image[j + 1];
85 if(minor) *minor = (char)image.image[j + 2];
86 }
87 else
88 {
89 if(major) *major = 0;
90 if(minor) *minor = 0;
91 }
92 return (eImageSignature)(1 << image_type); // Found all the chars required
93 }
94 }
95 }
96
97 }
98
99 // // Backup for old (16K) bootloader image
100 // if (strstr(filename.c_str(), is_samx70_bootloader_needle))
101 // {
102 // return IS_IMAGE_SIGN_ISB_SAMx70_16K;
103 // }
104
105 return IS_IMAGE_SIGN_NONE;
106}

108eImageSignature cISBootloaderBase::get_bin_image_signature(std::string filename, uint8_t* major, char* minor)
109{
110 FILE* blfile = 0;
111
112#ifdef _MSC_VER
113
114 fopen_s(&blfile, filename.c_str(), "rb");
115
116#else
117
118 blfile = fopen(filename.c_str(), "rb");
119
120#endif
121
122 if (blfile == 0)
123 return IS_IMAGE_SIGN_NONE;
124
125 fseek(blfile, 0x5FFC, SEEK_SET);
126 unsigned char ver_info[4];
127 fread(ver_info, 1, 4, blfile);
128 fclose(blfile);
129
130 //Check for marker for valid version info
131 if (ver_info[0] == 0xAA && ver_info[1] == 0x55)
132 {
133 if(major) *major = ver_info[2];
134 if(minor) *minor = ver_info[3];
135 }
136 else
137 {
138 if(major) *major = 0;
139 if(minor) *minor = 0;
140
141 // Look in the old location for this info (v5 and earler)
142 fseek(blfile, 0x3DFC, SEEK_SET);
143 fread(ver_info, 1, 4, blfile);
144 fclose(blfile);
145
146 //Check for marker for valid version info
147 if (ver_info[0] == 0xAA && ver_info[1] == 0x55)
148 {
149 if(major) *major = ver_info[2];
150 if(minor) *minor = ver_info[3];
151 }
152 }
153
154 return (eImageSignature)(IS_IMAGE_SIGN_ISB_SAMx70_16K | IS_IMAGE_SIGN_ISB_SAMx70_24K);
155}

164eImageSignature cISBootloaderBase::get_image_signature(std::string filename, uint8_t* major, char* minor)
165{
166 const char * extension = cISBootloaderBase::get_file_ext(filename.c_str());
167
168 if(strcmp(extension, "bin") == 0)
169 {
170 return cISBootloaderBase::get_bin_image_signature(filename, major, minor);
171 }
172 else if(strcmp(extension, "hex") == 0)
173 {
174 return cISBootloaderBase::get_hex_image_signature(filename, major, minor);
175 }
176
177 return IS_IMAGE_SIGN_NONE;

262 uint32_t bl_uINS_3 = get_image_signature(filenames.bl_uINS_3.path, &major, &minor) & (IS_IMAGE_SIGN_ISB_SAMx70_16K | IS_IMAGE_SIGN_ISB_SAMx70_24K);
263 uint32_t fw_IMX_5 = get_image_signature(filenames.fw_IMX_5.path) & IS_IMAGE_SIGN_UINS_5;
264 uint32_t bl_IMX_5 = get_image_signature(filenames.bl_IMX_5.path, &major, &minor) & IS_IMAGE_SIGN_ISB_STM32L4;
265 uint32_t fw_EVB_2 = get_image_signature(filenames.fw_EVB_2.path) & (IS_IMAGE_SIGN_EVB_2_16K | IS_IMAGE_SIGN_EVB_2_24K);
266 uint32_t bl_EVB_2 = get_image_signature(filenames.bl_EVB_2.path, &major, &minor) & (IS_IMAGE_SIGN_ISB_SAMx70_16K | IS_IMAGE_SIGN_ISB_SAMx70_24K);
267
268 obj = new cISBootloaderISB(updateProgress, verifyProgress, statusfn, handle);
269 (obj)->m_port_name = std::string(handle->port);
270 device = (obj)->check_is_compatible();
271 if(device == IS_IMAGE_SIGN_ERROR)
272 {
273 delete obj;
274 }
275 else if(device)
276 { // Firmware for a device must be specified to update its bootloader
277 if (((device & IS_IMAGE_SIGN_APP) & fw_EVB_2) && ((device & IS_IMAGE_SIGN_ISB) & bl_EVB_2))
278 {
279 (obj)->isb_mightUpdate = true;
280 addMutex->lock();
281 contexts.push_back(obj);
282 *new_context = obj;
283 addMutex->unlock();
284 return IS_OP_CLOSED;
285 }
286 else if (((device & IS_IMAGE_SIGN_ISB) & bl_IMX_5) && ((device & IS_IMAGE_SIGN_APP) & fw_IMX_5))
287 {
288 (obj)->isb_mightUpdate = true;
289 addMutex->lock();
290 contexts.push_back(obj);
291 *new_context = obj;
292 addMutex->unlock();
293 return IS_OP_CLOSED;
294 }
295 else if (((device & IS_IMAGE_SIGN_ISB) & bl_uINS_3) && ((device & IS_IMAGE_SIGN_APP) & fw_uINS_3))
296 {
297 (obj)->isb_mightUpdate = true;
298 addMutex->lock();
299 contexts.push_back(obj);
300 *new_context = obj;
301 addMutex->unlock();
302 return IS_OP_CLOSED;
303 }
304 else
305 {
306 delete obj;
307 return IS_OP_CLOSED;
308 }
309
310 if((obj)->isb_mightUpdate)
311 {
312 if(major != 0 && minor != 0)
313 {
314 if(major < (obj)->m_isb_major)
315 {
316 (obj)->isb_mightUpdate = false;
317 }
318 else if(major == (obj)->m_isb_major)
319 {
320 if(minor < (obj)->m_isb_minor)
321 {
322 (obj)->isb_mightUpdate = false;
323 }
324 else if(minor == (obj)->m_isb_minor)
325 {
326 (obj)->isb_mightUpdate = false;
327 }
328
329 }
330 }
331 }
332 }
333 else
334 {
335 delete obj;
336 }
337
338 return IS_OP_OK;
