129is_operation_result cISBootloaderSAMBA::download_image(std::string filename)
130{
131 serial_port_t* port = m_port;
132
133 // https://github.com/atmelcorp/sam-ba/tree/master/src/plugins/connection/serial
134 // https://sourceforge.net/p/lejos/wiki-nxt/SAM-BA%20Protocol/
135 uint8_t buf[SAMBA_PAGE_SIZE];
136
137 SAMBA_ERROR_CHECK(erase_flash(), "(SAMBA) Failed to erase flash memory");
138
139 // try non-USB and then USB mode (0 and 1)
140 for(int isUSB = 0; isUSB < 2; isUSB++)
141 {
142 serialPortSleep(port, 250);
143 serialPortFlush(port);
144
145 // flush
146 serialPortWrite(port, (const uint8_t*)"#", 2);
147 serialPortReadTimeout(port, buf, sizeof(buf), 100);
148
149 FILE* file;
150#ifdef _MSC_VER
151 fopen_s(&file, filename.c_str(), "rb");
152#else
153 file = fopen(filename.c_str(), "rb");
154#endif
155
156 if (file == 0)
157 {
158 SAMBA_STATUS("(SAMBA) Unable to load bootloader file", IS_LOG_LEVEL_ERROR);
159 // serialPortClose(port);
160 return IS_OP_ERROR;
161 }
162
163 fseek(file, 0, SEEK_END);
164 int size = ftell(file);
165 fseek(file, 0, SEEK_SET);
166 checksum = 0;
167
168 if (size != SAMBA_BOOTLOADER_SIZE_24K)
169 {
170 SAMBA_STATUS("(SAMBA) Invalid or old (v5 or earlier) bootloader file", IS_LOG_LEVEL_ERROR);
171 // serialPortClose(port);
172 return IS_OP_ERROR;
173 }
174
175 if(isUSB == 0) SAMBA_STATUS("(SAMBA) Writing ISB bootloader...", IS_LOG_LEVEL_INFO);
176
177 uint32_t offset = 0;
178 size_t len;
179 while ((len = fread(buf, 1, SAMBA_PAGE_SIZE, file)) == SAMBA_PAGE_SIZE)
180 {
181 if (flash_erase_write_page(offset, buf, isUSB) != IS_OP_OK)
182 {
183 if (!isUSB) { offset = 0; break; } // try USB mode
184 SAMBA_STATUS("Failed to upload page", IS_LOG_LEVEL_ERROR);
185 // serialPortClose(port);
186 return IS_OP_ERROR;
187 }
188 for (uint32_t* ptr = (uint32_t*)buf, *ptrEnd = (uint32_t*)(buf + sizeof(buf)); ptr < ptrEnd; ptr++)
189 {
190 checksum ^= *ptr;
191 }
192 offset += SAMBA_PAGE_SIZE;
193
194 m_update_progress = (float)offset / (float)SAMBA_BOOTLOADER_SIZE_24K;
195 if (m_update_callback != 0)
196 {
197 m_update_callback(this, m_update_progress);
198 }
199 }
200 fclose(file);
201 if (offset != 0) break; // success!
202 }
203
204 return IS_OP_OK;
205}

207is_operation_result cISBootloaderSAMBA::erase_flash()
208{
209 SAMBA_STATUS("(SAMBA) Erasing flash memory...", IS_LOG_LEVEL_INFO);
210
211 // Erase 3 sectors of 16 pares each (8K)
212 if (write_word(0x400e0c04, 0x5a000207) == IS_OP_OK)
213 {
214 SLEEP_MS(200); // From datasheet, max time it could take
215 wait_eefc_ready(true);
216 }
217 if (write_word(0x400e0c04, 0x5a001207) == IS_OP_OK)
218 {
219 SLEEP_MS(200); // From datasheet, max time it could take
220 wait_eefc_ready(true);
221 }
222 if (write_word(0x400e0c04, 0x5a002207) == IS_OP_OK)
223 {
224 SLEEP_MS(200); // From datasheet, max time it could take
225 return wait_eefc_ready(true);
226 }
227
228 return IS_OP_ERROR;
229}

231uint32_t cISBootloaderSAMBA::get_device_info()
232{
233 serial_port_t* port = m_port;
234 uint8_t buf[SAMBA_PAGE_SIZE];
235
236 // Flush the bootloader command buffer
237 serialPortWrite(port, (const uint8_t*)"#", 2);
238 int count = serialPortReadTimeout(port, buf, sizeof(buf), 100);
239
240 // Set non-interactive mode and wait for response
241 count = serialPortWriteAndWaitFor(port, (const uint8_t*)"N#", 2, (const uint8_t*)"\n\r", 2);
242 if (!count)
243 { // Failed to handshake with bootloader
244 // serialPortClose(port);
245 return 0;
246 }
247
248 // Set flash mode register
249 SAMBA_ERROR_CHECK_SN(write_word(0x400e0c00, 0x04000600), "Failed to set flash mode register");
250
251 // Set flash command to STUS (start read unique signature)
252 SAMBA_ERROR_CHECK_SN(write_word(0x400e0c04, 0x5a000014), "Failed to command signature readout");
253
254 // Wait until EEFC.FSR.FRDY is cleared
255 SAMBA_ERROR_CHECK_SN(wait_eefc_ready(false), "Failed to clear flash ready bit");
256
257 // Read out the unique identifier
258 SAMBA_ERROR_CHECK_SN(read_word(0x00400000 + offsetof(manufacturing_info_t, serialNumber), &m_sn), "Failed to read UID word");
259
260 // Set flash command to SPUS (stop read unique identifier)
261 SAMBA_ERROR_CHECK_SN(write_word(0x400e0c04, 0x5a000015), "Failed to command stop signature readout");
262
263 return m_sn;
264}

266is_operation_result cISBootloaderSAMBA::read_word(uint32_t address, uint32_t* word)
267{
268 uint8_t buf[16];
269 int count = SNPRINTF((char*)buf, sizeof(buf), "w%08x,#", address);
270 if ((serialPortWrite(m_port, buf, count) == count) &&
271 (serialPortReadTimeout(m_port, buf, sizeof(uint32_t), SAMBA_TIMEOUT_DEFAULT) == sizeof(uint32_t)))
272 {
273 *word = (buf[3] << 24) | (buf[2] << 16) | (buf[1] << 8) | buf[0];
274 return IS_OP_OK;
275 }
276 return IS_OP_ERROR;
277}

287is_operation_result cISBootloaderSAMBA::wait_eefc_ready(bool waitReady)
288{
289 uint32_t status = 0;
290 for (int i = 0; i < 10; i++)
291 {
292 // EEFC.FSR.FRDY - Flash ready status
293 if(read_word(0x400e0c08, &status) == IS_OP_ERROR) continue;
294 if( waitReady && (status & 0x01)) return IS_OP_OK; // status is ready and we are waiting for ready
295 if(!waitReady && !(status & 0x01)) return IS_OP_OK; // status is not ready and we are waiting for not ready
296 serialPortSleep(m_port, 20);
297 }
298 return IS_OP_ERROR;
299}

301is_operation_result cISBootloaderSAMBA::write_uart_modem(uint8_t* buf, size_t len)
302{
303 int ret;
304 uint8_t eot = UART_XMODEM_EOT; // "X" comes from xModem, not sure where that came from
305 uint8_t answer;
306 xmodem_chunk_t chunk = {0};
307 chunk.block = 1;
308 chunk.start = UART_XMODEM_SOH;
309
310 // wait for ping from bootloader
311 do
312 {
313 if (serialPortRead(m_port, &answer, sizeof(uint8_t)) != sizeof(uint8_t))
314 {
315 return IS_OP_ERROR;
316 }
317 } while (answer != 'C');
318
319 // write up to one sector
320 while (len)
321 {
322 size_t z = 0;
323
324 z = _MIN(len, sizeof(chunk.payload));
325 memcpy(chunk.payload, buf, z);
326 memset(chunk.payload + z, 0xff, sizeof(chunk.payload) - z);
327
328 chunk.crc = SWAP16(crc16(chunk.payload, sizeof(chunk.payload)));
329 chunk.block_neg = 0xff - chunk.block;
330
331 ret = serialPortWrite(m_port, (const uint8_t*)&chunk, sizeof(xmodem_chunk_t));
332 if (ret != sizeof(xmodem_chunk_t))
333 {
334 return IS_OP_ERROR;
335 }
336
337 ret = serialPortReadTimeout(m_port, &answer, sizeof(uint8_t), SAMBA_TIMEOUT_DEFAULT);
338 if (ret != sizeof(uint8_t))
339 {
340 return IS_OP_ERROR;
341 }
342
343 switch (answer)
344 {
345 case UART_XMODEM_ACK:
346 chunk.block++;
347 len -= z;
348 buf += z;
349 break;
350 case UART_XMODEM_CAN:
351 return IS_OP_ERROR;
352 default:
353 case UART_XMODEM_NAK:
354 break;
355 }
356 }
357
358 ret = serialPortWrite(m_port, &eot, sizeof(uint8_t));
359 if (ret != sizeof(uint8_t))
360 {
361 return IS_OP_ERROR;
362 }
363 // serialPortReadChar(&ctx->handle.port, &eot);
364 return IS_OP_OK;
365}

367is_operation_result cISBootloaderSAMBA::flash_erase_write_page(size_t offset, uint8_t data[SAMBA_PAGE_SIZE], bool isUSB)
368{
369 uint16_t page = (uint16_t)(offset / SAMBA_PAGE_SIZE);
370 uint8_t buf[32];
371 int count;
372
373 // Copy data into latch buffer prior to write
374 if (isUSB)
375 {
376 count = SNPRINTF((char*)buf, sizeof(buf), "S%08x,%08x#",
377 (unsigned int)(SAMBA_FLASH_START_ADDRESS + offset),
378 (unsigned int)SAMBA_PAGE_SIZE);
379 serialPortWrite(m_port, buf, count);
380 serialPortWrite(m_port, data, SAMBA_PAGE_SIZE);
381 }
382 else
383 {
384 count = SNPRINTF((char*)buf, sizeof(buf), "S%08x,#", (unsigned int)(SAMBA_FLASH_START_ADDRESS + offset));
385 serialPortWrite(m_port, buf, count);
386
387 // send page data
388 if (write_uart_modem(buf, SAMBA_PAGE_SIZE) != IS_OP_OK)
389 {
390 return IS_OP_ERROR;
391 }
392 }
393
394 // EEFC.FCR - WP - copy latch buffers into flash
395 count = SNPRINTF((char*)buf, sizeof(buf), "W%08x,5a%04x01#", 0x400e0c04, page);
396 serialPortWrite(m_port, buf, count);
397
398 return wait_eefc_ready(true);
399}

401is_operation_result cISBootloaderSAMBA::verify_image(std::string filename)
402{
403 (void)filename; // Checksum is used instead of re-reading file
404
405 uint32_t checksum2 = 0;
406 uint32_t nextAddress;
407 uint8_t buf[SAMBA_PAGE_SIZE];
408 int count;
409
410 serialPortFlush(m_port);
411
412 SAMBA_STATUS("(SAMBA) Verifying ISB bootloader...", IS_LOG_LEVEL_INFO);
413
414 for (uint32_t address = SAMBA_FLASH_START_ADDRESS; address < (SAMBA_FLASH_START_ADDRESS + SAMBA_BOOTLOADER_SIZE_24K); )
415 {
416 serialPortFlush(m_port);
417 nextAddress = address + SAMBA_PAGE_SIZE;
418 serialPortFlush(m_port);
419 while (address < nextAddress)
420 {
421 count = SNPRINTF((char*)buf, sizeof(buf), "w%08x,#", address);
422 serialPortWrite(m_port, buf, count);
423 serialPortSleep(m_port, 1);
424 address += sizeof(uint32_t);
425 }
426 count = serialPortReadTimeout(m_port, buf, SAMBA_PAGE_SIZE, SAMBA_TIMEOUT_DEFAULT);
427 if (count == SAMBA_PAGE_SIZE)
428 {
429 for (uint32_t* ptr = (uint32_t*)buf, *ptrEnd = (uint32_t*)(buf + sizeof(buf)); ptr < ptrEnd; ptr++)
430 {
431 checksum2 ^= *ptr;
432 }
433 }
434 else return IS_OP_ERROR;
435
436 m_verify_progress = (float)(address - SAMBA_FLASH_START_ADDRESS) / (float)SAMBA_BOOTLOADER_SIZE_24K;
437 if (m_verify_callback != 0)
438 {
439 m_verify_callback(this, m_verify_progress);
440 }
441 }
442 if (checksum != checksum2) return IS_OP_ERROR;
443 return IS_OP_OK;
444}

455uint16_t cISBootloaderSAMBA::crc_update(uint16_t crc_in, int incr)
456{
457 uint16_t crc = crc_in >> 15;
458 uint16_t out = crc_in << 1;
459
460 if (incr) out++;
461 if (crc) out ^= UART_XMODEM_CRC_POLY;
462
463 return out;
464}

467uint16_t cISBootloaderSAMBA::crc16(uint8_t* data, uint16_t size)
468{
469 uint16_t crc, i;
470
471 for (crc = 0; size > 0; size--, data++)
472 {
473 for (i = 0x80; i; i >>= 1)
474 {
475 crc = crc_update(crc, *data & i);
476 }
477 }
478
479 for (i = 0; i < 16; i++) crc = crc_update(crc, 0);
480
481 return crc;
482}
