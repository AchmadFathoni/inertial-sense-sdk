162static inline uint16_t libusb_cpu_to_le16(const uint16_t x)
163{
164 union {
165 uint8_t b8[2];
166 uint16_t b16;
167 } _tmp;
168 _tmp.b8[1] = (uint8_t) (x >> 8);
169 _tmp.b8[0] = (uint8_t) (x & 0xff);
170 return _tmp.b16;
171}
